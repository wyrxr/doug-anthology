% --- CUSTOM COMMAND FOR AUTHORS (With ToC Entry) ---
\makeatletter
\newcommand{\chapterauthor}[1]{%
    % 1. Format the name on the actual page
    {\parindent0pt\vspace*{-25pt}%
    \linespread{1.1}\large\scshape#1%
    \par\nobreak\vspace*{35pt}}
    \@afterheading%
    
    % 2. Inject the name into the Table of Contents
    \addtocontents{toc}{%
        \protect\setlength{\parindent}{1em} % Ensures no weird indentation
        {\itshape #1}\par       % Prints the name in Italics
        \protect\vspace{10pt}  % Adds a little space after the name
    }%
}
\makeatother
